-- DROP FUNCTION staging.excel_to_ddmmyy(numeric);

CREATE OR REPLACE FUNCTION staging.excel_to_ddmmyy(excel_date numeric)
 RETURNS text
 LANGUAGE plpgsql
 IMMUTABLE
AS $function$
BEGIN
    IF excel_date IS NULL OR excel_date <= 0 THEN
        RETURN NULL;
    END IF;
    RETURN TO_CHAR(
        TIMESTAMP '1899-12-30 00:00:00' + (excel_date * INTERVAL '1 day'),
        'DD/MM/YY'
    );
END;
$function$
;

-- DROP FUNCTION staging.excel_to_ddmmyy_time(numeric);

CREATE OR REPLACE FUNCTION staging.excel_to_ddmmyy_time(excel_date numeric)
 RETURNS text
 LANGUAGE plpgsql
 IMMUTABLE
AS $function$
BEGIN
    IF excel_date IS NULL OR excel_date <= 0 THEN
        RETURN NULL;
    END IF;
    RETURN TO_CHAR(
        TIMESTAMP '1899-12-30 00:00:00' + (excel_date * INTERVAL '1 day'),
        'DD/MM/YY HH24:MI'
    );
END;
$function$
;

-- DROP FUNCTION staging.excel_to_timestamp(numeric);

CREATE OR REPLACE FUNCTION staging.excel_to_timestamp(excel_date numeric)
 RETURNS timestamp without time zone
 LANGUAGE plpgsql
 IMMUTABLE
AS $function$
BEGIN
    IF excel_date IS NULL OR excel_date <= 0 THEN
        RETURN NULL;
    END IF;
    -- Excel epoch is 1899-12-30 (accounting for Excel's leap year bug)
    -- The integer part is days, decimal part is time of day
    RETURN TIMESTAMP '1899-12-30 00:00:00' + (excel_date * INTERVAL '1 day');
END;
$function$
;

-- DROP FUNCTION staging.format_date_ddmmyy(timestamp);

CREATE OR REPLACE FUNCTION staging.format_date_ddmmyy(ts timestamp without time zone)
 RETURNS text
 LANGUAGE plpgsql
 IMMUTABLE
AS $function$
BEGIN
    IF ts IS NULL THEN
        RETURN NULL;
    END IF;
    RETURN TO_CHAR(ts, 'DD/MM/YY');
END;
$function$
;
